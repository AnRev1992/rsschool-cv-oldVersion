Towel sort 
module.exports = function towelSort (matrix) {
    let arr = [];
    if (matrix === undefined || matrix.length === 0) {
        return []
    }
    for (let i = 0; i < matrix.length; i++) {
        for (let j = 0; j < matrix[i].length; j++) {
            let columIdx = i % 2 === 0 
            ? j
            : (matrix[i].length - j - 1);
            arr.push(matrix[i][columIdx]); 
        }
    }
    return arr;
}

function decode(expr) {
    let letter = '';
    let sliceLetter = '';
    let result = '';
                                        
    for (let i = 0; i < expr.length; i = i + 10) {
        let resultInDot = '';
        letter = expr.slice(i, i+10);
        branch: { 
            for (let j = 0; j < letter.length; j++) {
                if (letter[j] === '1') {
                    sliceLetter = letter.slice(j);
                    break branch;
                }
            }
        }
        if (letter === '**********') {
            result = result + ' ';
            sliceLetter='00';
        }
        for (let k = 0; k < sliceLetter.length; k = k + 2) {
            let dotResult = '';
            if (sliceLetter.slice(k, k+2) === '10') {
                dotResult = '.';                    
            }
            else if (sliceLetter.slice(k, k+2) === '11') {
                dotResult = '-';
            }
            if (dotResult !== '') {
                resultInDot = resultInDot + dotResult;
            }
        }
        if (resultInDot !== '') { 
            result = result + MORSE_TABLE[resultInDot];
        }
    }
    return result;
}
